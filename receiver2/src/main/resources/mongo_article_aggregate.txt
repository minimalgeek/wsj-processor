db.article.aggregate(
[
    {$unwind : "$articleWords"},
    {$group : 
        {
           _id : {year: { $year: "$dateTime" }, month: { $month: "$dateTime" }, day: { $dayOfMonth: "$dateTime" }, word: "$articleWords" },
           differentDocuments: {$sum : 1}
        }
    },
    {$sort : { differentDocuments : -1 }},
    {$out : "result"},
], {
    allowDiskUse:true,
    cursor:{}}
);
